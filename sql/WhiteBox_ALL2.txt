CREATE TABLE USERS(
    USR_ID NVARCHAR2(33) PRIMARY KEY,
    USR_PASSWORD CHAR(60) NOT NULL,
    USR_ENABLED CHAR(1) DEFAULT '1',
    USR_LASTNAME NVARCHAR2(500) NOT NULL,
    USR_FIRSTNAME NVARCHAR2(500) NOT NULL,
    USR_PHONE NVARCHAR2(500) NOT NULL,
    USR_EMAIL NVARCHAR2(500) NOT NULL,
    USR_ADDRESS NVARCHAR2(500) NOT NULL,
    USR_IMAGE NVARCHAR2(500) NOT NULL,
    USR_DATE DATE DEFAULT SYSDATE
);
CREATE TABLE AUTHORITIES (
    AUT_USRID NVARCHAR2(33),
    AUT_AUTHORITY VARCHAR2(20),
    PRIMARY KEY (AUT_USRID, AUT_AUTHORITY),
    FOREIGN KEY (AUT_USRID) REFERENCES USERS (USR_ID)
);
CREATE TABLE MEMBER (
    MEM_USRID NVARCHAR2(33) PRIMARY KEY,
    MEM_POSTAL NVARCHAR2(500) NOT NULL,
    MEM_GENDER CHAR(1) CHECK(MEM_GENDER IN ('M', 'F')) NOT NULL,
    MEM_BIRTHDAY DATE NOT NULL,
    MEM_ISSOCIAL CHAR(1) CHECK(MEM_ISSOCIAL IN ('T', 'F')) NOT NULL,
    FOREIGN KEY (MEM_USRID) REFERENCES USERS (USR_ID)
);
CREATE TABLE SELLER (
    SEL_USRID NVARCHAR2(33) PRIMARY KEY,
    SEL_SHOPNAME NVARCHAR2(20) NOT NULL,
    SEL_INFO NVARCHAR2(2000) NOT NULL,
    FOREIGN KEY (SEL_USRID) REFERENCES USERS (USR_ID)
);
CREATE TABLE AUTHENTICATIONRESULT (
    AUR_CODE CHAR(2) PRIMARY KEY, 
    AUR_NAME NVARCHAR2(10)
    /* 
    AS 인증코드 발송
    AN 인증 완료 (알림 용)
    AC 완료 확인 (알림 중단)
    */
);
CREATE TABLE AUTHENTICATIONLOG (
    AUL_USRID NVARCHAR2(33),
    AUL_DATE DATE DEFAULT SYSDATE,
    AUL_AURCODE CHAR(2) DEFAULT 'AS',
    PRIMARY KEY (AUL_USRID, AUL_DATE, AUL_AURCODE),
    FOREIGN KEY (AUL_USRID) REFERENCES USERS (USR_ID),
    FOREIGN KEY (AUL_AURCODE) REFERENCES AUTHENTICATIONRESULT (AUR_CODE)
);

CREATE TABLE ACCESSLOG (
    ASL_USRID NVARCHAR2(33) NOT NULL,
    ASL_DATE DATE DEFAULT SYSDATE,
    ASL_PUBLICIP VARCHAR(15) NOT NULL,
    ASL_PRIVATEIP VARCHAR(15) NOT NULL,
    ASL_LOCATION NVARCHAR2(500) NOT NULL,
    ASL_ACTION NUMBER(1) CHECK(ASL_ACTION IN(1, -1)),
    PRIMARY KEY (ASL_USRID, ASL_DATE, ASL_PUBLICIP, ASL_PRIVATEIP, ASL_LOCATION, ASL_ACTION),
    FOREIGN KEY (ASL_USRID) REFERENCES USERS (USR_ID)
);
CREATE TABLE FRIENDSTATE(
    FST_CODE CHAR(2) PRIMARY KEY,
    FST_NAME NVARCHAR2(10) NOT NULL
    /*
    FS 수락 대기
    FA 수락 완료
    FC 완료 확인
    */
);
CREATE TABLE FRIEND(
    FRI_SENDER NVARCHAR2(33),
    FRI_RECEIVER NVARCHAR2(33),
    FRI_DATE DATE DEFAULT SYSDATE,
    FRI_STATE CHAR(2) DEFAULT 'FS',
    PRIMARY KEY(FRI_SENDER, FRI_RECEIVER),
    FOREIGN KEY(FRI_SENDER) REFERENCES MEMBER(MEM_USRID),
    FOREIGN KEY(FRI_RECEIVER) REFERENCES MEMBER(MEM_USRID),
    FOREIGN KEY(FRI_STATE) REFERENCES FRIENDSTATE(FST_CODE)
);
CREATE TABLE FOLLOW (
    FOL_USRMEMID NVARCHAR2(33),
    FOL_USRSELID NVARCHAR2(33),
    PRIMARY KEY (FOL_USRMEMID, FOL_USRSELID), 
    foreign key (FOL_USRMEMID) REFERENCES MEMBER (MEM_USRID),
    foreign key (FOL_USRSELID) REFERENCES SELLER (SEL_USRID)
);   
CREATE TABLE CATEGORY (
    CAT_CODE CHAR(2) PRIMARY KEY,
    CAT_NAME NVARCHAR2(10) NOT NULL
    /*
    CC 패션
    CE 전자기기
    CB 생일
    CM 결혼/집들이
    CK 출산/키즈
    CL 명품
    */
);
CREATE TABLE PRODUCT (
    PRO_USRSELID NVARCHAR2(33),
    PRO_CODE NUMBER(4),
    PRO_CATCODE CHAR(2) NOT NULL,
    PRO_NAME NVARCHAR2(20) NOT NULL,
    PRO_PRICE NUMBER(7) DEFAULT -1,
    PRO_STOCK NUMBER(5) DEFAULT -1,
    PRO_INFO NVARCHAR2(2000) NOT NULL,
    PRO_ISDELETED CHAR(1) CHECK (PRO_ISDELETED IN ('T', 'F')),
    PRIMARY KEY (PRO_USRSELID, PRO_CODE),
    FOREIGN KEY (PRO_USRSELID) REFERENCES SELLER (SEL_USRID),
    FOREIGN KEY (PRO_CATCODE) REFERENCES CATEGORY (CAT_CODE)
);
CREATE TABLE PRODUCTIMAGE (
   PIG_PROUSRSELID NVARCHAR2(33),
   PIG_PROCODE NUMBER(4),
   PIG_CODE NUMBER(2),
   PIG_NAME NVARCHAR2(50),
   PIG_PATH NVARCHAR2(300),
   PIG_ISTHUMBNAIL CHAR(1) CHECK (PIG_ISTHUMBNAIL IN ('T', 'F')),
   PRIMARY KEY(PIG_PROUSRSELID, PIG_PROCODE, PIG_CODE),
   FOREIGN KEY(PIG_PROUSRSELID, PIG_PROCODE) 
        REFERENCES PRODUCT(PRO_USRSELID, PRO_CODE)
);
CREATE TABLE VIEWHISTORY(
    VHS_USRMEMID NVARCHAR2(33),
    VHS_PROUSRSELID  NVARCHAR2(33),
    VHS_PROCODE NUMBER(4),
    VHS_DATE DATE DEFAULT SYSDATE,
    VHS_ISVIEWED CHAR(1) CHECK (VHS_ISVIEWED IN ('T', 'F')),
    VHS_ISWATCHED CHAR(1) CHECK (VHS_ISWATCHED IN ('T', 'F')),
    VHS_ISWISHED CHAR(1) CHECK (VHS_ISWISHED IN ('T', 'F')),
    PRIMARY KEY (VHS_USRMEMID, VHS_PROUSRSELID, VHS_PROCODE),
    FOREIGN KEY (VHS_USRMEMID)
        REFERENCES MEMBER (MEM_USRID),
    FOREIGN KEY (VHS_PROUSRSELID, VHS_PROCODE)
        REFERENCES PRODUCT (PRO_USRSELID, PRO_CODE)
);
CREATE TABLE ORDERSTATE (
    OST_CODE CHAR(2) PRIMARY KEY,
    OST_NAME NVARCHAR2(10) NOT NULL
    /*
    OC 장바구니
    OO 주문 완료
    RF 환불 요청
    RA 환불 완료
    RC 환불 확인
    */
);
CREATE TABLE PAYMENTTYPE (
    PAY_CODE CHAR(2) PRIMARY KEY,
    PAY_NAME NVARCHAR2(10) NOT NULL
    /*
    PN 미결제
    PC 신용카드
    PK 카카오페이
    PP 페이팔
    */
);
CREATE TABLE ORDERS (
    ORD_DATE DATE DEFAULT SYSDATE,
    ORD_USRMEMID NVARCHAR2(33),
    ORD_OSTSTATE CHAR(2) DEFAULT 'OC',
    ORD_PAYCODE CHAR(2) DEFAULT 'PN',
    PRIMARY KEY (ORD_DATE, ORD_USRMEMID),
    FOREIGN KEY (ORD_USRMEMID) REFERENCES MEMBER(MEM_USRID),
    FOREIGN KEY (ORD_OSTSTATE) REFERENCES ORDERSTATE(OST_CODE),
    FOREIGN KEY (ORD_PAYCODE) REFERENCES PAYMENTTYPE(PAY_CODE)
);
CREATE TABLE GIFTSTATE(
    GST_CODE CHAR(2) PRIMARY KEY,
    GST_NAME NVARCHAR2(10) NOT NULL
    /*
    GS 수락 대기
    GA 수락 완료
    GC 완료 확인
    */
);
CREATE TABLE ORDERDETAIL (
    ODT_ORDDATE DATE DEFAULT SYSDATE,
    ODT_ORDMEMID NVARCHAR2(33),
    ODT_PROUSRSELID NVARCHAR2(33),
    ODT_PROCODE NUMBER(4),
    ODT_QTY NUMBER(2) DEFAULT -1,
    ODT_MEMRECIPIENT NVARCHAR2(33),
    ODT_GSTCODE CHAR(2) DEFAULT 'GS',
    PRIMARY KEY (ODT_ORDDATE, ODT_ORDMEMID, ODT_PROUSRSELID, ODT_PROCODE, ODT_MEMRECIPIENT), 
    FOREIGN KEY (ODT_ORDDATE, ODT_ORDMEMID) REFERENCES ORDERS (ORD_DATE, ORD_USRMEMID),
    FOREIGN KEY (ODT_PROUSRSELID, ODT_PROCODE) REFERENCES PRODUCT (PRO_USRSELID, PRO_CODE),
    FOREIGN KEY (ODT_MEMRECIPIENT) REFERENCES MEMBER (MEM_USRID),
    FOREIGN KEY (ODT_GSTCODE) REFERENCES GIFTSTATE (GST_CODE)
);

CREATE TABLE REACTION (
    REC_CODE CHAR(2) PRIMARY KEY,
    REC_NAME NVARCHAR2(50)
    /*
    RR 받고 싶어요
    RM 감동이에요
    RI 재미있어요
    RU 속상해요
    */
);
CREATE TABLE REVIEW(
    REV_ODTORDDATE DATE,
    REV_ODTORDMEMID NVARCHAR2(33),
    REV_ODTPROUSRSELID NVARCHAR2(33),
    REV_ODTPROCODE  NUMBER(4),
    REV_ODTMEMRECIPIENT NVARCHAR2(33),
    REV_RECCODE CHAR(2),
    REV_CONTENT NVARCHAR2(2000),
    REV_RATING NUMBER(1) CHECK (REV_RATING BETWEEN 1 AND 5),
    REV_REPLY NVARCHAR2(2000),
    REV_DATE DATE DEFAULT SYSDATE,
    PRIMARY KEY (
        REV_ODTORDDATE,
        REV_ODTORDMEMID,
        REV_ODTPROUSRSELID,
        REV_ODTPROCODE
        ),
    FOREIGN KEY (REV_ODTORDDATE, REV_ODTORDMEMID, REV_ODTPROUSRSELID, REV_ODTPROCODE, REV_ODTMEMRECIPIENT)
        REFERENCES ORDERDETAIL (ODT_ORDDATE, ODT_ORDMEMID, ODT_PROUSRSELID, ODT_PROCODE, ODT_MEMRECIPIENT),
    FOREIGN KEY (REV_RECCODE)
        REFERENCES REACTION (REC_CODE)
);

CREATE OR REPLACE TRIGGER ORDERS_UPDATE_TRG
    AFTER UPDATE OF ORD_DATE ON ORDERS
        FOR EACH ROW
        BEGIN
            UPDATE ORDERDETAIL
                SET ODT_ORDDATE = :NEW.ORD_DATE
                WHERE ODT_ORDDATE = :OLD.ORD_DATE
                AND ODT_ORDMEMID = :OLD.ORD_USRMEMID;
        END;
